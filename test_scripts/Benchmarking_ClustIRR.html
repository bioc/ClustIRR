Benchmarking

Create input data

An artificial clonal expansion of 100 vs 1 sequences of a made up CDR3b sequence is created as input.


# set seed for reproducibility
set.seed(1234)

## create artificial reference repertoire
data_ref_f <- immuneSIM(
  number_of_seqs = 600,         # number of simulated sequences
  species = "hs",               # species (human)
  receptor = "tr",              # receptor-type (t-cells)
  chain = "b",                  # chain type (beta chain)
  max_cdr3_length = 100,        # max cdr3 length
  min_cdr3_length = 6,          # min cdr3 length
  user_defined_alpha = 1.1,     # clone count distribution
  verbose = TRUE)               # output progress
#> initializing sim..
#> simulated sequences: 100 
#> simulated sequences: 200 
#> simulated sequences: 300 
#> simulated sequences: 400 
#> simulated sequences: 500 
#> simulated sequences: 600

## create artificial sample repertoire by inserting motifs

# # create n random motifs of length k
# motifs <- list(
#   "n" = 3,                      # 3 different motifs
#   "k" = 3,                      # 3mer's
#   "freq" = c(0.05, 0.25, 0.4))  # occurrence probabilities

# create motifs manually
motifs <- data.frame(aa   = c("CCC","FFF"),
                    nt   = c("gccgcc","tttttt"),
                    freq = c(0.2,0.1))

data_sample_f <- motif_implantation(
  sim_repertoire = data_ref_f,  # source repertoire
  motif = motifs,               # motifs to implant
  fixed_pos = 5)                # insert at random (0) or fixed (n) position
#> Joining by: aa

# random subset motif enriched dataset
data_sample_f <- data.frame(data_sample_f[sample(1:nrow(data_sample_f),
                                             50,
                                             replace = FALSE),])

# Reduce to CDR3b sequence
data_ref    <- data.frame(data_ref_f[,4])
data_sample <- data.frame(data_sample_f[,4])
names(data_ref)[1]    <- "CDR3b"
names(data_sample)[1] <- "CDR3b"

## Insert global sequence for global check

# Random sequence for global similarity
groundtruth_expansion <- c(CDR3b = "QQQQTTTQQQQTTT")

# Insert 1 time in data_ref for reference
data_ref <- rbind(data_ref, groundtruth_expansion) 

# Insert 20 times in sample
for(i in 0:19) {
  data_sample <- rbind(data_sample, groundtruth_expansion)
}

rm(i, groundtruth_expansion)

Prepare input parameters


cores <- parallel::detectCores()

# gliphR v1 + v2 + v3
ks <- c(2, 3, 4)
B <- 1000
control_input <- list(
  B = B,
  global_max_dist = 1,
  local_max_fdr = 0.05,
  local_min_ove = 2,
  local_min_o = 3,
  trim_flank_aa = 3,
  low_mem = FALSE,
  global_pairs = NULL)
